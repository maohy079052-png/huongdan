<script>
    // Cơ sở dữ liệu kịch bản mới: Hành trình trải nghiệm Chợ quê Tân Thuận Đông
    const scriptTree = {
        'start': {
            text: 'David: "Wow! Is this the famous Tan Thuan Dong countryside market? The atmosphere here is so rustic and amazing! What should we do first to experience this river culture?"',
            img: 'cong_cho.jpg',
            keywords: {
                'boat': 'boat_scene',
                'rowing': 'boat_scene',
                'walk': 'walk_scene',
                'inside': 'walk_scene'
            },
            options: [
                "Let's take a traditional rowing boat",
                "Let's walk inside to see the food stalls"
            ]
        },
        'boat_scene': {
            text: 'David: "This is beautiful! Rowing along the small canal feels so peaceful. Look at those fruit orchards! Oh, I can smell local food from here. What should we eat now?"',
            img: 'ghe_xuong.jpg',
            keywords: {
                'cake': 'cake_scene',
                'cakes': 'cake_scene',
                'coconut': 'coconut_scene',
                'water': 'coconut_scene'
            },
            options: [
                "We can try Vietnamese traditional cakes",
                "We can drink fresh local coconut water"
            ]
        },
        'walk_scene': {
            text: 'David: "Oh wow! Look at all these stalls! People are wearing traditional Ba Ba shirts and selling everything on banana leaves. Everything looks so clean and organic. What should we taste first?"',
            img: 'quay_hang.jpg',
            keywords: {
                'cake': 'cake_scene',
                'cakes': 'cake_scene',
                'coconut': 'coconut_scene',
                'water': 'coconut_scene'
            },
            options: [
                "Let's try some traditional sweet cakes",
                "Let's grab a cold fresh coconut water"
            ]
        },
        'cake_scene': {
            text: 'David: "Mmm! This pig-skin cake (Bánh da lợn) and banana cake are so soft and sweet! Eating these while listening to Don Ca Tai Tu music is just perfect. Thank you for this genuine countryside market tour!" 🎋🥞🎉',
            img: 'quay_banh.jpg',
            keywords: {}, 
            options: []
        },
        'coconut_scene': {
            text: 'David: "Ah! This local coconut is so sweet, refreshing, and perfectly cold! It completely washes away the heat. This eco-tour on the islet is unforgettable. Thanks a lot!" 🥥🛶🎉',
            img: 'trai_dua.jpg',
            keywords: {}, 
            options: []
        }
    };

    let currentStage = 'start';

    // Cấu hình Webkit Speech Recognition
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    let recognition = null;

    if (!SpeechRecognition) {
        alert("Thiết bị này cần mở bằng Google Chrome (Android) hoặc Safari (iPhone) để dùng chức năng nhận diện giọng nói!");
    } else {
        recognition = new SpeechRecognition();
        recognition.lang = 'en-US'; 
        recognition.interimResults = false;
        recognition.maxAlternatives = 1;
    }

    const startBtn = document.getElementById('start-btn');
    const btnText = document.getElementById('btn-text');
    const btnIcon = document.getElementById('btn-icon');
    const statusText = document.getElementById('status-text');
    const transcriptText = document.getElementById('transcript-text');
    const sceneryImg = document.getElementById('scenery-img');
    const storyText = document.getElementById('story');
    const optionsList = document.getElementById('options-list');
    const interactionBox = document.getElementById('interaction-box');
    const instructionTitle = document.getElementById('instruction-title');

    if (recognition) {
        startBtn.addEventListener('click', () => {
            try {
                recognition.start();
            } catch(e) {
                console.log("Hệ thống nhận diện đang hoạt động.");
            }
        });

        recognition.onstart = () => {
            statusText.className = "status-active";
            statusText.innerText = "🎙️ Đang nghe... Hãy đọc to câu thoại gợi ý bằng tiếng Anh.";
            btnText.innerText = "Hệ thống đang lắng nghe...";
            btnIcon.innerText = "🛑";
            startBtn.disabled = true;
        };

        recognition.onspeechend = () => {
            recognition.stop();
            startBtn.disabled = false;
            btnText.innerText = "Bắt đầu nói (Chạm để nói)";
            btnIcon.innerText = "🎙️";
        };

        recognition.onresult = (event) => {
            const resultText = event.results[0][0].transcript.toLowerCase().replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,"").trim();
            transcriptText.innerText = `"${event.results[0][0].transcript}"`;
            
            let isMatched = false;
            const currentKeywords = scriptTree[currentStage].keywords;

            for (let key in currentKeywords) {
                if (resultText.includes(key)) {
                    currentStage = currentKeywords[key];
                    updateApplicationState();
                    isMatched = true;
                    break;
                }
            }

            if (isMatched) {
                statusText.className = "status-success";
                statusText.innerText = "✅ Khớp từ khóa thành công! Chuyển cảnh.";
            } else {
                statusText.className = "status-error";
                statusText.innerText = "❌ Chưa khớp câu thoại gợi ý. Bạn hãy đọc lại rõ ràng hơn nhé!";
            }
        };

        recognition.onerror = (event) => {
            startBtn.disabled = false;
            btnText.innerText = "Bắt đầu nói (Chạm để nói)";
            btnIcon.innerText = "🎙️";
            statusText.className = "status-error";
            statusText.innerText = "Lỗi kết nối Micro: " + event.error;
        };
    }

    // Hàm cập nhật trạng thái ứng dụng
    function updateApplicationState() {
        const stage = scriptTree[currentStage];
        storyText.innerText = stage.text;
        sceneryImg.src = stage.img;
        optionsList.innerHTML = '';
        
        if (stage.options && stage.options.length > 0) {
            instructionTitle.style.display = 'block';
            instructionTitle.innerHTML = "🔊 Nhấn nút dưới rồi đọc TO câu gợi ý:";
            startBtn.style.display = 'inline-flex';
            
            stage.options.forEach(option => {
                const div = document.createElement('div');
                div.className = 'option-line';
                div.innerHTML = `👉 <strong>${option}</strong>`;
                optionsList.appendChild(div);
            });
        } else {
            // Fix lỗi ở code cũ: Ẩn title hướng dẫn và nút nói, giữ nguyên giao diện để bấm reset được
            instructionTitle.innerHTML = `<span style="color: var(--success-color); font-weight:bold;">🎉 Bạn đã hoàn thành chuyến dẫn khách trải nghiệm Chợ quê xuất sắc!</span>`;
            startBtn.style.display = 'none';
            statusText.className = "status-success";
            statusText.innerText = "Chuyến tham quan kết thúc tốt đẹp.";
        }
    }

    // Thiết lập tính năng chơi lại từ đầu
    function resetGame() {
        currentStage = 'start';
        transcriptText.innerText = "...";
        statusText.className = "";
        statusText.innerText = "Sẵn sàng. Nhấn nút để kích hoạt microphone.";
        updateApplicationState();
    }

    // Tự động chạy khi tải trang xong
    window.addEventListener('DOMContentLoaded', () => {
        updateApplicationState();

        // Tự động phát âm thanh giới thiệu (david.wav)
        const introAudio = new Audio('david.wav');
        introAudio.play().catch(() => {
            const playAudioOnInteraction = () => {
                introAudio.play();
                document.removeEventListener('click', playAudioOnInteraction);
                document.removeEventListener('touchstart', playAudioOnInteraction);
            };
            document.addEventListener('click', playAudioOnInteraction);
            document.addEventListener('touchstart', playAudioOnInteraction);
        });
    });
</script>
